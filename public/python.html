Typing + async

python 
from typing import Any

async def fetch_json(url: str) -> dict[str, Any]:
    import aiohttp
    async with aiohttp.ClientSession() as s:
        async with s.get(url) as r:
            r.raise_for_status()
            return await r.json()

Classes (init + property)

python 
class User:
    def __init__(self, email: str):
        self._email = email

    @property
    def email(self) -> str:
        return self._email
